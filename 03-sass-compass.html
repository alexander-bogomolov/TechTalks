SASS vs. SCSS

SASS

.class
  margin: 10px
  .subclass
    color: red

SCSS

.class {
  margin: 10px;
  .subclass {
    color: red;
  }
}

Variablen

SCSS

$logo-blue: #0090D1;
$logo-gray: #767372;

a { color: $logo-blue; }
a:hover { color: $logo-gray; }

CSS

a { color: #0090D1; }
a:hover { color: #767372; }

Rechnen

div { width: 40px + 2px - 5px / 2 * 10; }
$width-page: 960px;
$width-sidebar: 200px;
$width-content: $width-page - $width-sidebar;

#page { width: $width-page; }
#sidebar { width: $width-sidebar; }
#content { width: $width-content; }

Berechnugnen

round(3.3); /* 3 */
ceil(3.3); /* 4 */
floor(3.9);  /* 3 */
abs(-12); /* 12 */
precentage(26/50); /* 52 */

Nesting

SCSS

.class {
  color: green;
  .subclass {
    color: red;
  }
}

CSS

.class { color: green; }
.class .subclass { color: red; }

Nesting

SCSS

a {
  color: green;
  &:hover {
    color: red;
  }
}

button {
  background: #ccc;
  .no-bg & {
    background: transparent;
  }
}

CSS

a { color: green; }
a:hover { color: red; }

button { background: #ccc; }
.no-bg button { background: transparent; }

Auslagern

SCSS

.message {
  border: 1px solid transparent;
  padding: 1em;
  margin: 1em 0;
}
.error {
  @extend .message;
  border-color: red;
}

CSS

.message, .error {
  border: 1px solid transparent;
  padding: 1em;
  margin: 1em 0;
}
.error {
  border-color: red;
}

Mixins

SCSS

@mixin border-radius($radius:5px) {
  -webkit-border-radius: $radius;
     -moz-border-radius: $radius;
          border-radius: $radius;
}
div {
  @include border-radius(10px);
}

CSS

div {
  -webkit-border-radius: 10px;
     -moz-border-radius: 10px;
          border-radius: 10px;
}

Abfragen

SCSS

@mixin color($dark:false) {
  @if $dark == treu {
    color: #000;
    &:hover { color: blue; }
  } @else {
    color: #fff;
    &:hover { color: #ccc; };
  }
}
a.odd { @include color(); }
a.even { @include color(true); }

CSS

a.odd { color: #000 }
a.odd:hover { color: blue; }
a.even { color: #fff; }
a.even:hover { color: #ccc; }

Arbeiten mit Bildern

SCSS

header {
  background: image-url('mein-bild.png');
  h1 {
    width: image-width('mein-bild.png');
    height: image-height('mein-bild.png');
  }
}

CSS

header { background: url('images/mein-bild.png?123456789'); }
header h1 {
  width: 560px;
  height: 240px;
}

Sprites

SCSS

@import "icon/*.png";
@import all-icon-sprites($dimensions:true);

CSS

.icon-sprite, .icon-minus { background: url('images/icon-sd6115b5d.png') no-repeat; }
.icon-minus {
  background-position: 15px 255px;
  width: 16px;
  height: 16px;
}
